#### 链表

**要实现链表，我们首先要实现 Node （节点），每个节点都包含两份信息，一个是数据信息，另一个是指向下一节点的引用。**

```python
class Node(object):
    def __init__(self, init_data):
        self.data = init_data
        self.next = None

    def get_data(self):
        return self.data

    def get_next(self):
        return self.next

    def set_data(self, new_data):
        self.data = new_data

    def set_next(self, new_next):
        self.next = new_next

```

##### 无序链表

```python
# 链表只提供一个入口（头部），所以其他节点只能通过头部和 next 来进行访问
class UnorderedList(object):
    def __init__(self):
        self.head = None

    def is_empty(self):
        return self.head is None

    # 新元素作为 最后一个元素
    def add(self, item):
        if self.head is None:
            self.head = Node(item)
        else:
            current = self.head
            while current is not None:
                if current.get_next() is None:
                    new_node = Node(item)
                    current.set_next(new_node)
                    break
                current = current.get_next()

    # 新元素作为第一个元素
    def add(self, item):
        temp = Node(item)
        temp.set_next(self.head)
        self.head = temp

    def length(self):
        current = self.head
        count = 0
        while current is not None:
            count = count + 1
            current = current.get_next()
        return count

    def search(self, item):
        current = self.head
        found = False

        while (current is not None) and (not found):
            if current.get_data() == item:
                found = True
            else:
                current = current.get_next()

        return found

    def remove(self, item):
        current = self.head
        pre = None
        found = False
        while not found:
            if current.get_data() == item:
                found = True
            else:
                pre = current
                current = current.get_next()

        if pre is None:
            self.head = current.get_next()
        else:
            pre.set_next(current.get_next())

    def __str__(self):
        unordered_list = ""

        current = self.head

        while current is not None:
            unordered_list = unordered_list + "->" + f"{current.get_data()}"
            current = current.get_next()

        return unordered_list[2:]


if __name__ == '__main__':
    u = UnorderedList()
    u.add(12)
    u.add(43)
    u.add(10)

    print(u)
    u.remove(43)
    print(u)


```

##### 有序链表

```python
# 链表只提供一个入口（头部），所以其他节点只能通过头部和 next 来进行访问
class OrderedList(object):
    def __init__(self):
        self.head = None

    def is_empty(self):
        return self.head is None

    # 新元素作为第一个元素
    def add(self, item):
        current = self.head
        pre = None

        while current is not None:
            if current.get_data() > item:
                break
            else:
                pre = current
                current = current.get_next()

        temp = Node(item)
        # 说明链表是空的，元素作为第一个元素
        if pre is None:
            temp.set_next(self.head)
            self.head = temp
        # 说明 current 是 None ，或者 current 是比 item 第一个大的元素.
        else:
            temp.set_next(current)
            pre.set_next(temp)

    def length(self):
        current = self.head
        count = 0
        while current is not None:
            count = count + 1
            current = current.get_next()
        return count

    def search(self, item):
        current = self.head
        found = False

        while (current is not None) and (found is False):
            if current.get_data() == item:
                found = True
                break
            if current.get_data() > item:
                break
            current = current.get_next()
        return found

    def remove(self, item):
        current = self.head
        pre = None
        found = False
        while not found:
            if current.get_data() == item:
                found = True
            else:
                pre = current
                current = current.get_next()

        if pre is None:
            self.head = current.get_next()
        else:
            pre.set_next(current.get_next())

    def __str__(self):
        ordered_list = ""

        current = self.head

        while current is not None:
            ordered_list = ordered_list + "->" + f"{current.get_data()}"
            current = current.get_next()

        return ordered_list[2:]


if __name__ == '__main__':
    o = OrderedList()
    o.add(4)
    o.add(2)
    o.add(3)

    print(o)
```
